# The following contains a set of instructions for setting up an Apache proxy for the LSP services



# -----------------------------
# Create a VM (Openstack/..)
# -----------------------------

..

# Create a floating IP and assign to the VM



# -----------------------------
# Install Apache
# -----------------------------
sudo apt-get install -y apache2
sudo apt-get install -y libapache2-mod-proxy-html libxml2-dev
  # for ubuntu bionic:  sudo apt-get install -y libapache2-mod-proxy-uwsgi-dbg libapache2-mod-proxy-uwsgi

sudo apache2ctl configtest
sudo a2enmod proxy
sudo a2enmod proxy_http
sudo a2enmod proxy_balancer
sudo a2enmod lbmethod_byrequests
sudo a2enmod proxy rewrite proxy_http proxy_wstunnel




# -----------------------------
# Configure Apache
# -----------------------------


sudo nano /etc/apache2/apache2.conf
   ..
   ServerName IP_ADDRESS
   ..


sudo nano /etc/apache2/sites-available/000-default.conf
..

<VirtualHost *:80>
        # The ServerName directive sets the request scheme, hostname and port that
        # the server uses to identify itself. This is used when creating
        # redirection URLs. In the context of virtual hosts, the ServerName
        # specifies what hostname must appear in the request's Host: header to
        # match this virtual host. For the default virtual host (this file) this
        # value is not decisive as it is used as a last resort host regardless.
        # However, you must set it for any further virtual host explicitly.
        #ServerName www.example.com
        ServerName 128.232.227.162

        ServerAdmin webmaster@localhost
        DocumentRoot /var/www/html

        # Available loglevels: trace8, ..., trace1, debug, info, notice, warn,
        # error, crit, alert, emerg.
        # It is also possible to configure the loglevel for particular
        # modules, e.g.
        #LogLevel info ssl:warn

        ErrorLog ${APACHE_LOG_DIR}/error.log
        CustomLog ${APACHE_LOG_DIR}/access.log combined

        # For most configuration files from conf-available/, which are
        # enabled or disabled at a global level, it is possible to
        # include a line for only one particular virtual host. For example the
        # following line enables the CGI configuration for this host only
        # after it has been globally disabled with "a2disconf".
        #Include conf-available/serve-cgi-bin.conf

        ProxyRequests Off
        ProxyPreserveHost On

        ProxyTimeout 6000
        Timeout 6000
        ProxyReceiveBufferSize 2048


        ## PROXY ENTRIES HERE ##
        ProxyPassMatch    ^/firefly\/(.*)$  http://192.168.0.13:32159/suit/$1
        ProxyPassReverse  ^/firefly\/(.*)$  http://192.168.0.13:32159/suit/$1

        ProxyPassMatch    ^/firefly  http://192.168.0.13:32159/suit
        ProxyPassReverse  ^/firefly  http://192.168.0.13:32159/suit

        ProxyPassMatch    ^/suit\/(.*)$  http://192.168.0.13:32159/suit/$1
        ProxyPassReverse  ^/suit\/(.*)$  http://192.168.0.13:32159/suit/$1

        ProxyPassMatch    ^/portal\/app/\/(.*)$  http://192.168.0.13:32159/suit/$1
        ProxyPassReverse  ^/portal/\app/\/(.*)$  http://192.168.0.13:32159/suit/$1

        ProxyPassMatch    ^/portal/suit\/(.*)$  http://192.168.0.13:32159/suit/$1
        ProxyPassReverse  ^/portal/suit\/(.*)$  http://192.168.0.13:32159/suit/$1

        ProxyPassMatch    ^/js9\/(.*)$  http://192.168.0.13:30557/$1
        ProxyPassReverse  ^/js9\/(.*)$  http://192.168.0.13:30557/$1

        ProxyPassMatch    ^/wf\/(.*)$  http://192.168.0.13:31292/$1
        ProxyPassReverse  ^/wf\/(.*)$  http://192.168.0.13:31292/$1

        ProxyPassMatch    ^/nublado\/(.*)$  http://192.168.0.13:31089/$1
        ProxyPassReverse  ^/nublado\/(.*)$  http://192.168.0.13:31089/$1

        ProxyPassMatch    ^/nb\/(.*)$  http://192.168.0.13:31089/nb/$1
        ProxyPassReverse  ^/nb\/(.*)$  http://192.168.0.13:31089/nb/$1

        ProxyPassMatch    ^/nb(.*)$  http://192.168.0.13:31089/nb/$1
        ProxyPassReverse  ^/nb(.*)$  http://192.168.0.13:31089/nb/$1

        ProxyPassMatch    ^/tap\/(.*)$  http://192.168.0.13:31348/tap/$1
        ProxyPassReverse  ^/tap\/(.*)$  http://192.168.0.13:31348/tap/$1

        ProxyPassMatch    ^/api\/(.*)$  http://192.168.0.13:31348/api/$1
        ProxyPassReverse  ^/api\/(.*)$  http://192.168.0.13:31348/api/$1
     
        ProxyPassMatch    ^/landingpage\/(.*)$  http://192.168.0.13:30505/$1
        ProxyPassReverse  ^/landingpage\/(.*)$  http://192.168.0.13:30505/$1

        ProxyPassMatch    ^/(.*)$  http://192.168.0.13:30505/$1
        ProxyPassReverse  ^/(.*)$  http://192.168.0.13:30505/$1

        ProxyPassMatch    ^/static/(.*)$  http://192.168.0.13:30505/$1
        ProxyPassReverse  ^/static/(.*)$  http://192.168.0.13:30505/$1


        RewriteEngine On
        RewriteCond %{HTTP:Upgrade} =websocket [NC]
        RewriteRule ^/firefly\/(.*)           ws://192.168.0.13:32159/$1 [P,L]
        RewriteCond %{HTTP:Upgrade} !=websocket [NC]
        RewriteRule ^/firefly\/(.*)           http://192.168.0.13:32159/$1 [P,L]

        RewriteEngine On
        RewriteCond %{HTTP:Upgrade} =websocket [NC]
        RewriteRule ^/suit\/(.*)           ws://192.168.0.13:32159/suit/$1 [P,L]
        RewriteCond %{HTTP:Upgrade} !=websocket [NC]
        RewriteRule ^/suit\/(.*)           http://192.168.0.13:32159/suit/$1 [P,L]

        RewriteEngine On
        RewriteCond %{HTTP:Upgrade} =websocket [NC]
        RewriteRule ^/portal/suit\/(.*)           ws://192.168.0.13:32159/suit/$1 [P,L]
        RewriteCond %{HTTP:Upgrade} !=websocket [NC]
        RewriteRule ^/portal/suit\/(.*)           http://192.168.0.13:32159/suit/$1 [P,L]

        RewriteEngine On
        RewriteCond %{HTTP:Upgrade} =websocket [NC]
        RewriteRule ^/nb\/(.*)           ws://192.168.0.13:31089/nb/$1 [P,L]
        RewriteCond %{HTTP:Upgrade} !=websocket [NC]





</VirtualHost>

..

# -----------------------------
# Restart Apache
# -----------------------------


sudo systemctl restart apache2
